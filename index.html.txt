const fixDatabaseTo2026 = (questionsArray) => {
    questionsArray.forEach(q => {
        const text = q.question.toLowerCase();
        if (text.includes("stirna") && text.includes("limitēt")) {
            const noIdx = q.options.findIndex(o => o.toLowerCase().includes("nē") || o.toLowerCase().includes("nav limitēt"));
            if (noIdx !== -1) q.answer = noIdx;
        }
        if (text.includes("meža cūka") && (text.includes("limitēt") || text.includes("termiņ"))) {
            const yearIdx = q.options.findIndex(o => o.toLowerCase().includes("visu gadu") || o.toLowerCase().includes("nē"));
            if (yearIdx !== -1) q.answer = yearIdx;
        }
        if (text.includes("lūsis")) {
            const forbiddenIdx = q.options.findIndex(o => o.toLowerCase().includes("nav medījam") || o.toLowerCase().includes("nē"));
            if (forbiddenIdx !== -1) q.answer = forbiddenIdx;
        }
    });
};

fixDatabaseTo2026(allQuestions);

// --- RENDERĒŠANA ---
const listContainer = document.getElementById('questionsList');
const modeToggle = document.getElementById('modeToggle');

function renderQuestions(filter = "") {
    listContainer.innerHTML = "";
    let visibleCount = 0;

    allQuestions.forEach(q => {
        if (q.question.toLowerCase().includes(filter.toLowerCase())) {
            visibleCount++;
            const qDiv = document.createElement('div');
            qDiv.className = 'question-block';
            let optionsHtml = q.options.map((opt, i) => `
                <div class="opt ${i === q.answer ? 'correct' : ''}" onclick="handleOptClick(this, ${i === q.answer})">
                    ${opt}
                </div>
            `).join('');
            qDiv.innerHTML = `<span class="q-title">${q.id}. ${q.question}</span><div class="options-list">${optionsHtml}</div>`;
            listContainer.appendChild(qDiv);
        }
    });
    document.getElementById('counter').innerText = `Atrasti: ${visibleCount}`;
}

function handleOptClick(element, isCorrect) {
    if (!modeToggle.checked) {
        element.classList.add(isCorrect ? 'revealed-correct' : 'revealed-wrong');
        element.innerText = (isCorrect ? "✅ " : "❌ ") + element.innerText.replace("✅ ", "").replace("❌ ", "");
    }
}

modeToggle.addEventListener('change', () => {
    document.body.className = modeToggle.checked ? "show-answers" : "hide-answers";
    renderQuestions(document.getElementById('searchInput').value);
});

document.getElementById('searchInput').addEventListener('input', (e) => renderQuestions(e.target.value));

window.onscroll = function() {
    document.getElementById("scrollTop").style.display = (window.scrollY > 300) ? "block" : "none";
};

window.onload = () => renderQuestions();
